124 inline void LData::restoreArrays()
125 {
126  int ierr;
127  if (d_ghosted_local_array)
128  {
129  ierr = VecRestoreArray(d_ghosted_local_vec, &d_ghosted_local_array);
130  IBTK_CHKERRQ(ierr);
131  d_ghosted_local_array = NULL;
132  delete d_boost_ghosted_local_array;
133  delete d_boost_vec_ghosted_local_array;
134  d_boost_ghosted_local_array = NULL;
135  d_boost_vec_ghosted_local_array = NULL;
136  }
137  if (d_ghosted_local_vec)
138  {
139  ierr = VecGhostRestoreLocalForm(d_global_vec, &d_ghosted_local_vec);
140  IBTK_CHKERRQ(ierr);
141  d_ghosted_local_vec = NULL;
142  }
143  if (d_array)
144  {
145  ierr = VecRestoreArray(d_global_vec, &d_array);
146  IBTK_CHKERRQ(ierr);
147  d_array = NULL;
148  delete d_boost_array;
149  delete d_boost_local_array;
150  delete d_boost_vec_array;
151  delete d_boost_local_vec_array;
152  d_boost_array = NULL;
153  d_boost_local_array = NULL;
154  d_boost_vec_array = NULL;
155  d_boost_local_vec_array = NULL;
156  }
157  return;
158 } // restoreArray

176 inline void LData::getArrayCommon()
177 {
178  if (!d_array)
179  {
180  int ierr = VecGetArray(d_global_vec, &d_array);
181  IBTK_CHKERRQ(ierr);
182  int ilower, iupper;
183  ierr = VecGetOwnershipRange(d_global_vec, &ilower, &iupper);
184  IBTK_CHKERRQ(ierr);
185  typedef boost::multi_array_types::extent_range range;
186  if (d_depth == 1)
187  {
188  typedef boost::multi_array<double, 1> array_type;
189  array_type::extent_gen extents;
190  d_boost_array =
191  new boost::multi_array_ref<double, 1>(d_array, extents[range(ilower, iupper)]);
192  d_boost_local_array =
193  new boost::multi_array_ref<double, 1>(d_array, extents[iupper - ilower]);
194  }
195  typedef boost::multi_array<double, 2> array_type;
196  array_type::extent_gen extents;
197  d_boost_vec_array = new boost::multi_array_ref<double, 2>(
198  d_array, extents[range(ilower / d_depth, iupper / d_depth)][d_depth]);
199  d_boost_local_vec_array = new boost::multi_array_ref<double, 2>(
200  d_array, extents[(iupper - ilower) / d_depth][d_depth]);
201  }
202  return;
203 } // getArrayCommon

205 inline void LData::getGhostedLocalFormArrayCommon()
206 {
207  if (!d_ghosted_local_vec)
208  {
209  int ierr = VecGhostGetLocalForm(d_global_vec, &d_ghosted_local_vec);
210  IBTK_CHKERRQ(ierr);
211  }
212  if (!d_ghosted_local_array)
213  {
214  int ierr = VecGetArray(d_ghosted_local_vec, &d_ghosted_local_array);
215  IBTK_CHKERRQ(ierr);
216  int ilower, iupper;
217  ierr = VecGetOwnershipRange(d_ghosted_local_vec, &ilower, &iupper);
218  IBTK_CHKERRQ(ierr);
219  if (d_depth == 1)
220  {
221  d_boost_ghosted_local_array = new boost::multi_array_ref<double, 1>(
222  d_ghosted_local_array, boost::extents[iupper - ilower]);
223  }
224  d_boost_vec_ghosted_local_array = new boost::multi_array_ref<double, 2>(
225  d_ghosted_local_array, boost::extents[(iupper - ilower) / d_depth][d_depth]);
226  }
227  return;
228 } // getGhostedLocalFormArrayCommon
