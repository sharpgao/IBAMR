103 inline void FixedSizedStream::pack(const bool* data, const int n)
104 {
105  const int bytes = SAMRAI::tbox::AbstractStream::sizeofBool(n);
106  void* ptr = getPointerAndAdvanceCursor(bytes);
107  char* c_ptr = static_cast<char*>(ptr);
108  for (int i = 0; i < n; i++)
109  {
110  c_ptr[i] = (data[i] ? 1 : 0);
111  }
112  return;
113 } // pack
114 
115 inline void FixedSizedStream::unpack(bool* data, const int n)
116 {
117  const int bytes = SAMRAI::tbox::AbstractStream::sizeofBool(n);
118  void* ptr = getPointerAndAdvanceCursor(bytes);
119  const char* c_ptr = static_cast<const char*>(ptr);
120  for (int i = 0; i < n; i++)
121  {
122  data[i] = (c_ptr[i] == 1);
123  }
124  return;
125 } // unpack

299 /*
300 *************************************************************************
301 * *
302 * Packing/unpacking helper functions. The member function *
303 * getPointerAndAdvanceCursor() returns a pointer to buffer space and *
304 * advances internal pointers to reflect the allocated buffers space. *
305 * The two functions given below simplify packing and unpacking for *
306 * the * numerous member functions below. *
307 * *
308 *************************************************************************
309 */
310 
311 inline void* FixedSizedStream::getPointerAndAdvanceCursor(const int bytes)
312 {
313  void* ptr = &d_buffer[d_buffer_index];
314  d_buffer_index += bytes;
315  if (d_buffer_index > d_current_size)
316  {
317  d_current_size = d_buffer_index;
318  if (d_buffer_index > d_buffer_size)
319  {
320  TBOX_ERROR("FixedSizedStream::getPointerAndAdvanceCursor():\n"
321  << " buffer overrun." << std::endl);
322  }
323  }
324  return ptr;
325 } // getPointerAndAdvanceCursor

327 template <typename T>
328 inline void FixedSizedStream::__pack(const T* const m_data, unsigned int m_bytes)
329 {
330  void* const ptr = getPointerAndAdvanceCursor(m_bytes);
331  memcpy(ptr, static_cast<const void*>(m_data), m_bytes);
332  return;
333 } // _pack

335 template <typename T>
336 inline void FixedSizedStream::__unpack(T* const m_data, unsigned int m_bytes)
337 {
338  const void* const ptr = getPointerAndAdvanceCursor(m_bytes);
339  memcpy(static_cast<void*>(m_data), ptr, m_bytes);
340  return;
341 } // _unpack
